import React from 'react';
import cn from 'classnames';
import PropTypes from 'prop-types';
import checkImg from '../../../../assets/icons/check-white.svg';
 
import * as styles from './Checkbox.module.scss';
 
export default function Checkbox({className, isChecked}) {
    return (
        <div className={cn(styles.checkbox, {[styles.checked]: isChecked}, className)}>
            {
                isChecked && <img src={checkImg} className={styles.checkImg} alt=''/>
            }
        </div>
    )
}
 
Checkbox.propTypes = {
    className: PropTypes.string,
    isChecked: PropTypes.bool
};
