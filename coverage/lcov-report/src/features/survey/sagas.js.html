import { take, fork, all, put, call, select } from 'redux-saga/effects'
import { SubmissionError } from 'redux-form';
import * as constants from './constants'
import {getSurveyPageSuccess, getSurveyPageRequest, submitSurveyPageSuccess} from './actions';
import {callApi} from '../../libs/awsLib';
 
export function* getSurveyPage() {
    while (true) {
        const {page} = yield take(constants.GET_SURVEY_PAGE_REQUEST);
 
        const response = yield call(callApi, {path: `/survey-pages/leftoff`});
        const body = yield response.json();
        if (response.ok) {
            yield put(getSurveyPageSuccess(body))
        }
    }
}
 
export function* submitSurveyPage() {
    while (true) {
        const {payload: {values, reject}} = yield take(constants.SUBMIT_SURVEY_PAGE_REQUEST);
 
        const state = yield select();
        const currentPage = state.survey.currentPage;
        const totalPages = state.survey.totalPages;
        const answers = [];
        for (let key in values) {
            answers.push({
                questionId: key,
                textResponse: values[key].textResponse,
                choiceDefIds: values[key].choiceDefIds
            });
        }
 
        const response = yield call(callApi, {path: `/answers`, method: 'post', body: {answers}});
 
        if (response.ok) {
            yield put(submitSurveyPageSuccess());
            if (currentPage < totalPages) {
                yield put(getSurveyPageRequest(currentPage + 1));
            }
        } else {
            reject(new SubmissionError({_error: 'Error'}));
        }
    }
}
 
export function* cancelSurvey() {
    while (true) {
        const {payload: {history}} = yield take(constants.CANCEL_SURVEY_REQUEST);
 
        const response = yield call(callApi, {path: '/cancel-survey', method: 'PUT', body: {}});
        if (response.ok) {
            history.push('/survey');
        }
    }
}
 
function startSagas(...sagas) {
    return function* rootSaga() {
        yield all(sagas.map(saga => fork(saga)))
    }
}
 
export default startSagas(getSurveyPage, submitSurveyPage, cancelSurvey)
