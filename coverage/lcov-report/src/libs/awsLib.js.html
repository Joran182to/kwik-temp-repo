import AWS from 'aws-sdk';
 
import {
  AuthenticationDetails,
  CognitoUser,
  CognitoUserAttribute,
  CognitoUserPool,
} from 'amazon-cognito-identity-js';
 
import requestSigner from './requestSigner';
 
const HEADER_AGENT_ID = 'x-spr-agent-id';
const HEADER_CLIENT_ID = 'x-spr-client-id';
const HEADER_GROUP_ID = 'x-spr-group-id';
const HEADER_USER_ID = 'x-spr-user-id';
const STORAGE_REMEMBER_ME = 'spr-remember-me';
const STORAGE_USER_ATTRIBUTES = 'spr-user-attributes';
const USER_ID_PREFIX = 'spr:user::';
 
export async function callApi({
  body,
  headers = {},
  method = 'GET',
  path,
  queryParams = {},
}) {
  if (!await isAuthenticated()) throw new Error('User is not logged in.');
 
  // attach spraoi secific headers
  const userAttributes = getUserAttributes();
  headers[HEADER_CLIENT_ID] = process.env.REACT_APP_SPR_CLIENT_ID;
  headers[HEADER_USER_ID] = userAttributes.user_id;
 
  if (userAttributes.group_id) {
    headers[HEADER_GROUP_ID] = userAttributes.group_id;
  }
 
  if (userAttributes.agent_id) {
    headers[HEADER_AGENT_ID] = userAttributes.agent_id;
  }
  
  let testUrl = window.location.hash.substring(1);
  let signedRequest = requestSigner.newClient({
    accessKey: AWS.config.credentials.accessKeyId,
    secretKey: AWS.config.credentials.secretAccessKey,
    sessionToken: AWS.config.credentials.sessionToken,
    region: process.env.REACT_APP_API_REGION,
    endpoint: testUrl || process.env.REACT_APP_API_URL,
  });
 
  if (typeof signedRequest.signRequest === 'function') {
    signedRequest = signedRequest.signRequest(
      { method, path, headers, queryParams, body });
 
    body = body ? JSON.stringify(body) : body;
    headers = signedRequest.headers;
 
    const results = await fetch(signedRequest.url, { method, headers, body });
    if (!results.ok) throw new Error(await results.text());
    return results;
  } else {
    throw new Error('An unknown error occurred. Please refresh the page.')
  }
}
 
export function completeSignUp({
  callbacks,
  newAttributes,
  newPassword,
  oldAttributes,
  user,
}) {
  return user.completeNewPasswordChallenge(
    newPassword,
    generateAttributeArray(newAttributes, oldAttributes),
    callbacks
  );
}
 
export function getNewUser(email) {
  return new CognitoUser({ Username: email, Pool: getNewUserPool() });
}
 
export function getUserAttributes() {
  return window[STORAGE_USER_ATTRIBUTES]
    || JSON.parse(localStorage.getItem(STORAGE_USER_ATTRIBUTES));
}
 
export async function isAuthenticated() {
  if (
    AWS.config.credentials &&
    Date.now() < AWS.config.credentials.expireTime - 60000
  ) {
    return true;
  }
 
  const currentUser = getNewUserPool().getCurrentUser();
  if (!currentUser) return false;
 
  if (isRemembered()) {
    const userToken = await getUserToken(currentUser);
    await getAwsCredentials(userToken);
    return true;
  } else {
    signOut();
    return false;
  }
}
 
export function isRemembered() {
  return localStorage.getItem(STORAGE_REMEMBER_ME) === 'true'
    || sessionStorage.getItem(STORAGE_REMEMBER_ME) === 'true';
}
 
export function rememberSession(rememberMe) {
  localStorage.setItem(STORAGE_REMEMBER_ME, rememberMe);
  sessionStorage.setItem(STORAGE_REMEMBER_ME, true);
}
 
export function resetPassword({ user, verificationCode, newPassword }) {
  return new Promise((resolve, reject) => {
    user.confirmPassword(verificationCode, newPassword, {
      onSuccess: data => resolve(data),
      onFailure: err => reject(err),
    });
  });
}
 
export function sendResetPasswordCode(email) {
  const user = getNewUser(email);
 
  return new Promise((resolve, reject) => {
    user.forgotPassword({
      onSuccess: data => resolve(user),
      onFailure: err => reject(err),
    });
  });
}
 
export function setUserAttributes(attributes) {
  attributes.user_id = USER_ID_PREFIX + attributes.sub;
  window[STORAGE_USER_ATTRIBUTES] = attributes;
  localStorage.setItem(STORAGE_USER_ATTRIBUTES, JSON.stringify(attributes));
}
 
export function signIn({
  email,
  newPasswordRequired = () => {},
  password,
  rememberMe = false,
  user,
}) {
  return new Promise((resolve, reject) => {
    const callbacks = {
      onSuccess: result => {
        setUserAttributes(result.idToken.payload);
        rememberSession(rememberMe);
        resolve(result);
      },
 
      onFailure: err => reject(err),
    };
 
    callbacks.newPasswordRequired = (userAttributes, requiredAttributes) =>
      newPasswordRequired(userAttributes, callbacks);
 
    user.authenticateUser(getAuthenticationDetails(email, password), callbacks);
  });
}
 
export function signOut() {
  const currentUser = getNewUserPool().getCurrentUser();
  if (currentUser !== null) currentUser.signOut();
  if (AWS.config.credentials) AWS.config.credentials.clearCachedId();
}
 
export function signUp({ email, password, attributes = {} }) {
  return new Promise((resolve, reject) =>
    getNewUserPool().signUp(
      email,
      password,
      generateAttributeArray(attributes),
      null,
      (err, result) => err ? reject(err) : resolve(result.user),
    ));
}
 
export function verifyEmail({ user, verificationCode }) {
  return new Promise((resolve, reject) =>
    user.confirmRegistration(
      verificationCode,
      true,
      (err, result) => err ? reject(err) : resolve(result)
    ));
}
 
function generateAttributeArray(data, attributes = []) {
  data['custom:client_id'] = process.env.REACT_APP_SPR_CLIENT_ID;
 
  for (const key in data) {
    attributes.push(new CognitoUserAttribute({ Name: key, Value: data[key] }));
  }
 
  return attributes;
}
 
function getAuthenticationDetails(email, password) {
  const authenticationData = { Username: email, Password: password };
  return new AuthenticationDetails(authenticationData);
}
 
function getAwsCredentials(userToken) {
  const authenticator = `cognito-idp.${process.env.REACT_APP_COGNITO_REGION}.` +
    `amazonaws.com/${process.env.REACT_APP_COGNITO_USER_POOL_ID}`;
 
  AWS.config.update({ region: process.env.REACT_APP_COGNITO_REGION });
 
  AWS.config.credentials = new AWS.CognitoIdentityCredentials({
    IdentityPoolId: process.env.REACT_APP_COGNITO_IDENTITY_POOL_ID,
    Logins: { [authenticator]: userToken },
  });
 
  return new Promise(resolve => AWS.config.credentials.refresh(resolve));
}
 
function getNewUserPool() {
  return new CognitoUserPool({
    UserPoolId: process.env.REACT_APP_COGNITO_USER_POOL_ID,
    ClientId: process.env.REACT_APP_COGNITO_APP_CLIENT_ID,
  });
}
 
function getUserToken(currentUser) {
  return new Promise((resolve, reject) => {
    currentUser.getSession((err, session) => {
      if (err) reject(err);
      else resolve(session.getIdToken().getJwtToken());
    });
  });
}
